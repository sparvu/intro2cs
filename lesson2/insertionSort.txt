# InsertionSort Procedure 
# ################################
InsertionSort (A)
   
    i, j
    n = length(A)

    # we advance the position through the entire array
    for i from 1 to n-1 
     
        ### assume the min is the first element 
        j = i
        
        while j > 0 and A[j-1] > A[j]
            swap A[j] with A[j-1]
	    j = j -1
	end while

	i = i +1
    
    # Array is now sorted
    return A

end procedure
